            ListItem.SETRANGE("List Code","List Code");
            ListItem.SETRANGE("Dimension Code","Dimension Code");

          END;

  }

    VAR
      Text000@1000 : TextConst 'DEU=Diese Funktion aktualisiert die EinrÅckung fÅr alle Zeilen der Liste %1.;ENU=This function updates the indentation of all the items for list %1.';
      Text001@1001 : TextConst 'DEU=Alle Zeilen zwischen "Beginn Summe" und dem zugehîrigen "Ende Summe" wurden um einen Level eingerÅckt.;ENU=All items between a Begin-Total and the matching End-Total are indented by one level.';

      ListItem@1006 : Record 6086004;
      Window@1007 : Dialog;
      i@1009 : Integer;

    PROCEDURE Indent@1();
    VAR
      NoOfItems@1000 : Integer;
      Progress@1001 : Integer;
    BEGIN
      Window.OPEN(Text004);

      NoOfItems := ListItem.COUNT;
      IF NoOfItems = 0 THEN
        NoOfItems := 1;
      WITH ListItem DO
        IF FIND('-') THEN
          REPEAT
            Progress := Progress + 1;
            Window.UPDATE(1,10000 * Progress DIV NoOfItems);

            IF "Account Type" = "Account Type"::"End-total" THEN BEGIN
              IF i < 1 THEN
                ERROR(
                  Text005,
                  "No.");
              i := i - 1;
            END;

            Indentation := i;
            MODIFY;

            IF "Account Type" = "Account Type"::"Begin-total" THEN
              i := i + 1;
          UNTIL NEXT = 0;

      Window.CLOSE;
    END;

    BEGIN
    END.
  }
}
