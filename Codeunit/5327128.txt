    BEGIN
      // ** MAIN FUNCTION THAT IS USED BY THE RECORD INDIVIDUAL FUNCTIONS BELOW
      // Get Filter from the Virtual Table that is defined in the current Web Page Element
      Mgt.Log('RecordFilter', 'BEGIN GetFilterFromVirtualTableOfWPE RecRef.NUMBER:'+FORMAT(RecRef.NUMBER));
      ModuleMapping.GetWebPageElement(WebPageElement);

        IF RecRef.NUMBER <> 0 THEN RecRef.CLOSE;

    END;

      Manages filters between Record References from User Interface and Virtual Tables to Records
      Records then can be used for Report.SETTABLEVIEW for Report printing in Events

      Get_<TableName>(RecRef,Record)     If current Web Page Element is based on a virtual table, the Record contains the filter of the
                                         values set by the user plus the ReadSecurityFilter by Events.
                                         If current Web Page Element is not based on a virtual table, the current RecRef is left untouched
                                         Example: Report Request Forms with one Record to filter on

      Get_VT<TableName>(VirtualTableCode,Record)
                                         Reads the Virtual Table on the current Web Page Element and sets the filter to the record
                                         plus the ReadSecurityFilter by Events.
                                         Example: Report Request Forms with several different Records to filter on

      SETTABLE<TableName>(RecRef,Record) Copies all filters and data of RecRef to Record
                                         (SETTABLE should do this, but is not working properly in NAV)
                                         Attention: Only the first 200 filter groups are used.
