    VAR
      HasValue@1001 : Boolean;
      Int@1002 : Integer;
      Dec@1003 : Decimal;
      D@1007 : Date;
      T@1008 : Time;
    BEGIN
      CASE TRUE OF
        Value.ISBOOLEAN :
          HasValue := Value;
        Value.ISOPTION :
          HasValue := TRUE;
        Value.ISINTEGER :
          BEGIN
            Int := Value;
            HasValue := Int <> 0;
          END;
        Value.ISDECIMAL :
          BEGIN
            Dec := Value;
            HasValue := Dec <> 0;
          END;
        Value.ISDATE :
          BEGIN
            D := Value;
            HasValue := D <> 0D;
          END;
        Value.ISTIME :
          BEGIN
            T := Value;
            HasValue := T <> 0T;
          END;
        Value.ISBINARY :
          HasValue := FALSE;
        ELSE
          HasValue := FORMAT(Value) <> '';
      END;
      EXIT(HasValue);
    END;
